This package supports creating an manipulating of vCard types. It follows the IETF specification as defined in http://www.ietf.org/rfc/rfc2426.txt

InputStream is = ...; // get a vCard as InputStream e.g. from a file
List myvcards = vCard.parsevCard(is);
IType e = null;
for (vCard vc : myvcards){
e = vc.getType(IType.FN);
if (e!=null)
System.out.println(e.getValue());
}
